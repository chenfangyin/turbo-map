üöÄ Quick Start

// Install
npm install turbo-map

// Import
import { TurboMap } from 'turbo-map';

// Create instance
const map = new TurboMap();

// Use complex objects as keys
const user = { id: 1, name: 'John', settings: { theme: 'dark' } };
map.set(user, 'user data');

// Retrieve value
const data = map.get(user); // 'user data'

// Check if key exists
const exists = map.has(user); // true

// Iterate over entries
for (const [key, value] of map) {
    console.log(key, value);
}
